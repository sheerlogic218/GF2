prog def all_gates : in.a len 1, in.b len 1 : out.y1 len 1, 
                     out.y2 len 1, out.y3 len 1;

    assign out.y1 = in.a + in.b;        (* OR  *)
    assign out.y2 = in.a * in.b;        (* AND *)
    assign out.y3 = in.a ^ in.b;        (* XOR *)

    monitor out.y1;
    monitor out.y2;
    monitor out.y3;

end def;
