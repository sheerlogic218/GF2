module complex_expr : -> ;
 
    switch A = 1 ;
    switch B = 0 ;
    switch C = 1 ;
    switch D = 0 ;
 
    wire out1 ;
    wire out2 ;
    wire out3 ;
 
    // Nested parentheses
    out1 = (A + B) * (C + D) ;
 
    // ! on a parenthesised expression (NOR equivalent)
    out2 = !(A + B) ;
 
    // Mixed: (A XOR B) OR (C AND D)
    out3 = (A ^ B) + (C * D) ;
 
    monitor out1 ;
    monitor out2 ;
    monitor out3 ;
 
end ;
